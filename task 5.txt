#include <iostream>
using namespace std;

int main() {
    double c, f;

    cout << "Celsius daxil et: ";
    cin >> c;
    cout << "Fahrenheit: " << c * 9/5 + 32 << endl;

    cout << "Fahrenheit daxil et: ";
    cin >> f;
    cout << "Celsius: " << (f - 32) * 5/9 << endl;

    return 0;
}
